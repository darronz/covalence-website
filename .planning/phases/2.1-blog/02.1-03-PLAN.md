```astro
import { getCollection, render } from 'astro:content';
import type { CollectionEntry } from 'astro:content';
import Base from '../../layouts/Base.astro';
import Nav from '../../components/Nav.astro';
import Footer from '../../components/Footer.astro';
import PostLayout from '../../components/PostLayout.astro';

export async function getStaticPaths() {
  const posts = await getCollection('posts');
  return posts.map((post) => ({
    params: { slug: post.id },
    props: { post },
  }));
}

interface Props {
  post: CollectionEntry<'posts'>;
}

const { post } = Astro.props;
const { Content } = await render(post);

const canonicalURL = new URL(`/posts/${post.id}/`, Astro.site);
const ogImageURL = post.data.og_image
  ? new URL(post.data.og_image, Astro.site)
  : new URL('/og-image.png', Astro.site);
const publishedISO = post.data.pub_date.toISOString();
const author = post.data.author ?? 'Darron Schall';
---

<Base title={`${post.data.title} — Covalence`} description={post.data.description}>
  <Fragment slot="head-extra">
    <meta property="og:type" content="article" />
    <meta property="article:published_time" content={publishedISO} />
    <meta property="article:author" content={author} />
    <meta property="og:image" content={ogImageURL} />
    <meta name="twitter:image" content={ogImageURL} />
    <link rel="canonical" href={canonicalURL} />
  </Fragment>
  <Nav />
  <PostLayout post={post}>
    <Content />
  </PostLayout>
  <Footer />
</Base>
```

Important — Base.astro slot extension. The frontmatter above uses `<Fragment slot="head-extra">` to inject per-post head tags. Base.astro (as shipped) does NOT currently expose a `head-extra` slot — it MUST be added in Plan 04 Task 3 (same plan, same file as the RSS `<link rel="alternate">` edit). In this task, implement the route as written; the `<Fragment slot="head-extra">` will be a no-op until Plan 04 ships. Astro ignores slot children whose target slot is undefined (no error, no warning).

If during execution that assumption proves wrong (build fails on unknown slot), fall back to emitting these metas inside `<PostLayout>` children (they'll render in the body, which OG scrapers tolerate less well but does not break the build). Document the chosen approach in the SUMMARY.

Full file contents (2-space indent, single quotes, semicolons, TypeScript strict):

```typescript
import rss from '@astrojs/rss';
import { getCollection } from 'astro:content';
import { marked } from 'marked';
import type { APIContext } from 'astro';

// Match releases.astro marked config so bodies render consistently across the site.
marked.setOptions({ gfm: true });

export async function GET(context: APIContext) {
  if (!context.site) {
    throw new Error(
      'rss.xml.ts: Astro `site` is not configured. Set `site` in astro.config.mjs.'
    );
  }

  const posts = (await getCollection('posts'))
    .sort((a, b) => b.data.pub_date.valueOf() - a.data.pub_date.valueOf())
    .slice(0, 50);

  const items = posts.map((post) => {
    let contentHTML = '';
    try {
      contentHTML = marked.parse(post.body ?? '') as string;
    } catch (err) {
      console.warn(`[rss] marked.parse failed for ${post.id}:`, err);
      contentHTML = `<p>${post.data.description}</p>`;
    }

    return {
      title: post.data.title,
      description: post.data.description,
      pubDate: post.data.pub_date,
      link: `/posts/${post.id}/`,
      content: contentHTML,
      author: post.data.author ?? 'Darron Schall',
    };
  });

  return rss({
    title: 'Covalence Blog',
    description: 'Long-form writing on Covalence — local-first memory, MCP, and the engineering behind it.',
    site: context.site,
    items,
    xmlns: { content: 'http://purl.org/rss/1.0/modules/content/' },
  });
}
```

Security constraints (mapped to phase threat model):
- T-02.1 item 1 (Markdown content trust): `marked` converts post body Markdown to HTML for `<content:encoded>`. `@astrojs/rss` wraps any string passed as `content` in a CDATA section (no further escaping), so Markdown-rendered HTML passes through to feed readers as-is. Since only the project owner authors posts (CLAUDE.md — single-committer repo), the trust assumption is "frontmatter + Markdown is trusted at author time". The `marked` library does NOT sanitise by default. If a post body contains raw `<script>` or attacker-supplied HTML snippets, they will propagate. Acceptance for Phase 2.1: document this in the threat model (below) and rely on author discipline + CF Pages preview review on every post before merge. DO NOT add `DOMPurify` or `sanitize-html` as deps in this phase — scope creep.
- T-02.1 item 2 (RSS URL construction): `link` is set to the relative path `/posts/${post.id}/` and `site: context.site` is passed explicitly. `@astrojs/rss` resolves each `link` against `site` to produce absolute URLs, so preview builds on `*.pages.dev` do NOT leak preview-host URLs into the feed.

Key contract notes:
- `context.site` is an Astro-provided `URL` object equal to the `site:` value in `astro.config.mjs` (`https://covalence.app`).
- `post.body` is the raw markdown string provided by Astro's glob loader. Available on each `CollectionEntry` returned from `getCollection`.
- The `xmlns: { content: '...' }` entry enables `<content:encoded>` emission per `@astrojs/rss` docs.

Do NOT:
- Implement manual XML string building — use the `rss()` helper from `@astrojs/rss` exclusively.
- Use Astro's container API (`experimental_AstroContainer`) to render `<Content />` — `marked.parse(post.body)` is the project's established markdown-to-HTML path (see `releases.astro` lines 6-8).
- Pass raw `post.body` as `content` — Markdown text in `<content:encoded>` without parsing would render as literal asterisks and backticks in feed readers.
- Cap at anything other than 50 (CONTEXT.md locks the value).
- Add `customData: '<language>en-GB</language>'` or other feed-level metadata — not specified in CONTEXT.md; keep the endpoint minimal. If that becomes a request after v1 ships, extend in a later phase.
- Use `getCollection('posts', (post) => !post.data.draft)` filter — CONTEXT.md locks out `draft` field.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| `post.data.og_image` (frontmatter) → HTML `<meta>` attribute | User-authored string interpolated into a URL and emitted as an OG/Twitter `<meta>` tag. Malicious value (e.g. `javascript:alert(1)`) could appear in social-card scrapers' output. |
| `post.body` (markdown) → RSS `<content:encoded>` HTML | Raw markdown parsed via `marked` (no sanitisation) and emitted into the RSS feed. Any raw HTML or `<script>` tag in the markdown body would propagate to feed readers. |
| `context.site` / `Astro.site` → RSS `link`, `/posts/<slug>/` canonical | Build-time constant (`https://covalence.app` per astro.config.mjs). Not derived from request host. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02.1-10 | Tampering / XSS | `og_image` frontmatter value in `[...slug].astro` | mitigate | Value is interpolated via `new URL(post.data.og_image, Astro.site)`. `new URL()` throws synchronously at build time if the string is not a valid absolute path / URL, causing the build to fail loudly. `javascript:alert(1)` fails this check (not a valid relative-to-base URL). This closes the T-02.1-04 concern from Plan 01. Post-Phase-3 enhancement: tighten Zod schema further. |
| T-02.1-11 | Information Disclosure | Preview-host URL in RSS feed | mitigate | `site: context.site` is passed explicitly to `rss()`. `context.site` comes from `astro.config.mjs` `site: 'https://covalence.app'`, NOT from a request header. CF Pages preview builds use the same config, so preview feeds also carry production-canonical URLs (acceptable — previews are not indexed). |
| T-02.1-12 | Tampering | Markdown body to RSS via `marked.parse` (no sanitisation) | accept | Posts are authored only by the project owner (single-committer repo). `marked` does NOT sanitise by default. Raw HTML / `<script>` tags in post markdown would propagate to feed readers. Mitigation relies on author discipline + mandatory CF Pages preview review before merge (Plan 05). If the project opens to external contributors, revisit with `sanitize-html` or `DOMPurify`. Documented as a known residual risk; not blocking for Phase 2.1. |
| T-02.1-13 | Tampering | Markdown body to post HTML page via Astro `<Content />` | mitigate | Astro's default markdown pipeline (used by `<Content />`) passes raw HTML through from markdown source by default but does not execute it server-side. Astro's `rehype` default config strips nothing but renders safely in browser context (`<script>` tags are executed client-side — same residual risk as T-02.1-12). Same mitigation: author discipline + preview review. |
| T-02.1-14 | Denial of Service | Markdown parse error in `marked.parse` for RSS | mitigate | Wrapped in `try/catch`; on parse failure the RSS item falls back to `<p>{description}</p>`. Build proceeds; a single malformed post does not break the feed for all posts. |
| T-02.1-15 | Tampering | Fragment `slot="head-extra"` in Base.astro | accept | Base.astro does not currently define this slot (Plan 04 adds it). Until then, per-post head tags are silently ignored by Astro. This is acceptable because (a) no posts exist yet when Plan 03 lands, and (b) Plan 04 lands the slot before any post merges. |

Block: `T-02.1-10` — `new URL()` validation is a hard build-time gate.
Accept (with documentation): `T-02.1-12`, `T-02.1-13` — classic static-site content-trust model; mitigation is review discipline, not code.
