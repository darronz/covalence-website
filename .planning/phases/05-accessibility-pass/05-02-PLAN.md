**Step 1: Calculate contrast ratios programmatically.**

Use a Node.js inline script to compute the WCAG 2.1 relative luminance and contrast ratio for each pairing. Run this in the terminal:

```bash
node -e "
function sRGB(c) { c /= 255; return c <= 0.04045 ? c / 12.92 : Math.pow((c + 0.055) / 1.055, 2.4); }
function luminance(hex) {
  const r = parseInt(hex.slice(1,3), 16);
  const g = parseInt(hex.slice(3,5), 16);
  const b = parseInt(hex.slice(5,7), 16);
  return 0.2126 * sRGB(r) + 0.7152 * sRGB(g) + 0.0722 * sRGB(b);
}
function contrast(hex1, hex2) {
  const l1 = luminance(hex1);
  const l2 = luminance(hex2);
  const lighter = Math.max(l1, l2);
  const darker = Math.min(l1, l2);
  return (lighter + 0.05) / (darker + 0.05);
}
const pairs = [
  ['#e8e6ed', '#08060d', 'text-primary on bg-primary', 4.5],
  ['#f0506a', '#08060d', 'accent on bg-primary (large text)', 3.0],
  ['#a78bfa', '#08060d', 'accent-subtle on bg-primary', 4.5],
  ['#1a0e00', '#f0506a', 'button text on accent', 4.5],
];
pairs.forEach(([fg, bg, label, threshold]) => {
  const ratio = contrast(fg, bg).toFixed(2);
  const pass = parseFloat(ratio) >= threshold;
  console.log(label + ': ' + ratio + ':1 ' + (pass ? 'PASS' : 'FAIL (need ' + threshold + ':1)'));
});
"
```

Use `npx axe` with a local server. Since this is a static site, serve it with `npx serve dist -l 4173` in the background, then scan:

```bash
npx serve dist -l 4173 &
sleep 2
npx axe http://localhost:4173/ http://localhost:4173/releases/ --tags wcag2a,wcag2aa --exit
```

The `--tags wcag2a,wcag2aa` flag restricts the scan to WCAG 2.0/2.1 Level A and AA rules.
