```typescript
interface Env {
  RELEASES_BUCKET: R2Bucket;
  RELEASE_ANALYTICS: AnalyticsEngineDataset;
}

// --- Shared analytics utilities ---

type FileType = 'dmg' | 'delta' | 'appcast' | 'release-notes' | 'other';

function classifyFile(name: string): FileType {
  if (name.endsWith('.dmg')) return 'dmg';
  if (name.endsWith('.delta')) return 'delta';
  if (name === 'appcast.xml') return 'appcast';
  if (name.endsWith('.md') || name.endsWith('.html')) return 'release-notes';
  return 'other';
}

function extractVersion(name: string): string {
  // Covalence-1.4.0.dmg -> "1.4.0"
  const dmgMatch = name.match(/Covalence-(\d+\.\d+\.\d+)/);
  if (dmgMatch) return dmgMatch[1];
  // Covalence10-8.delta -> "10-8"
  const deltaMatch = name.match(/Covalence(\d+-\d+)\.delta/);
  if (deltaMatch) return deltaMatch[1];
  return '';
}

function parseMacOSVersion(ua: string): string {
  const match = ua.match(/macOS\s+([\d.]+)/);
  return match ? match[1] : '';
}

interface AnalyticsParams {
  fileName: string;
  responseSize: number;
  request: Request;
}

function buildDataPoint(params: AnalyticsParams): AnalyticsEngineDataPoint {
  const { fileName, responseSize, request } = params;
  const ua = request.headers.get('user-agent') || '';
  const cf = (request as any).cf as IncomingRequestCfProperties | undefined;

  return {
    blobs: [
      fileName,                              // blob1: file name
      classifyFile(fileName),                // blob2: file type
      extractVersion(fileName),              // blob3: version
      cf?.country || '',                     // blob4: country
      ua.slice(0, 256),                      // blob5: user agent (truncated)
      parseMacOSVersion(ua),                 // blob6: macOS version
    ],
    doubles: [responseSize],                 // double1: response size bytes
    indexes: [fileName],                     // sampling key
  };
}
```

1. Create `functions/releases/[[path]].ts` — the catch-all handler for `/releases/*`:

```typescript
import { classifyFile, buildDataPoint } from '../analytics';

export const onRequest: PagesFunction<Env> = async (context) => {
  // Extract the path segments from the catch-all param
  const pathSegments = context.params.path;
  // [[path]] gives string | string[] — normalize
  const key = Array.isArray(pathSegments)
    ? pathSegments.join('/')
    : pathSegments || '';

  if (!key) {
    return new Response('Not Found', { status: 404 });
  }

  const object = await context.env.RELEASES_BUCKET.get(key);

  if (!object) {
    return new Response('Not Found', { status: 404 });
  }

  const headers = new Headers();
  object.writeHttpMetadata(headers);
  headers.set('ETag', object.httpEtag);

  // Cache releases for 24 hours — immutable artifacts (versioned filenames)
  headers.set('Cache-Control', 'public, max-age=86400');

  // Content-type fallback if R2 metadata doesn't have it
  if (!headers.has('Content-Type')) {
    if (key.endsWith('.dmg')) {
      headers.set('Content-Type', 'application/octet-stream');
    } else if (key.endsWith('.delta')) {
      headers.set('Content-Type', 'application/octet-stream');
    } else if (key.endsWith('.xml')) {
      headers.set('Content-Type', 'application/xml');
    } else if (key.endsWith('.html')) {
      headers.set('Content-Type', 'text/html; charset=utf-8');
    } else if (key.endsWith('.md')) {
      headers.set('Content-Type', 'text/markdown; charset=utf-8');
    }
  }

  // Non-blocking analytics write
  context.waitUntil(
    Promise.resolve().then(() => {
      context.env.RELEASE_ANALYTICS.writeDataPoint(
        buildDataPoint({
          fileName: key,
          responseSize: object.size,
          request: context.request,
        })
      );
    })
  );

  return new Response(object.body, { status: 200, headers });
};
```

import { buildDataPoint } from '../analytics';

export const onRequest: PagesFunction<Env> = async (context) => {
  const object = await context.env.RELEASES_BUCKET.get('appcast.xml');

  if (!object) {
    return new Response('Not Found', { status: 404 });

  const headers = new Headers();
  object.writeHttpMetadata(headers);
  headers.set('ETag', object.httpEtag);

  // Appcast checked frequently by Sparkle — cache 5 minutes
  headers.set('Cache-Control', 'public, max-age=300');
  headers.set('Content-Type', 'application/xml; charset=utf-8');

        buildDataPoint({
          fileName: 'appcast.xml',
          responseSize: object.size,
          request: context.request,
        })
      );
    })

Key implementation details:
- `context.params.path` in a `[[path]].ts` catch-all is `string | string[]` — always normalize.
- `object.writeHttpMetadata(headers)` copies content-type, content-disposition, etc. from R2 metadata to response headers.
- `object.httpEtag` is already RFC-9110 quoted — use it directly.
- `writeDataPoint()` returns void and writes in the background — wrapping in `waitUntil` ensures the data point is flushed even if the response completes first.
- The `Promise.resolve().then()` wrapper ensures `writeDataPoint` runs asynchronously in the `waitUntil` lifecycle.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Internet -> Pages Function | Untrusted HTTP requests from any client hit the functions |
| Pages Function -> R2 | Internal service binding (trusted, no auth needed) |
| Pages Function -> Analytics Engine | Internal service binding (trusted, no auth needed) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-n7h-01 | Spoofing | R2 key from URL path | mitigate | Key is derived from URL path segments only (no query params, no `../` traversal). CF Pages routing normalizes paths before they reach the function. The key is passed directly to `R2Bucket.get()` which is scoped to the bound bucket. |
| T-n7h-02 | Information Disclosure | Error responses | mitigate | Return generic "Not Found" for missing objects. Never expose R2 error details or bucket name in responses. |
| T-n7h-03 | Denial of Service | Analytics write volume | accept | Analytics Engine has built-in rate limiting. Each request writes exactly one data point. No amplification vector. |
| T-n7h-04 | Tampering | User-Agent / country spoofing | accept | Analytics data is observational, not security-critical. Spoofed UA or country does not affect access control. |
| T-n7h-05 | Information Disclosure | UA in analytics blob | accept | UA is already visible to CF edge. Storing truncated (256 char) UA in Analytics Engine adds no new exposure surface. |
</threat_model>
